try { // 1. Enviar para a API do Gemini const geminiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${GEMINI_API_KEY}`; const requestBody = { contents: [{ parts: [ { text: "Analise esta imagem de comida. Retorne APENAS um número inteiro representando a estimativa total de calorias (kcal). Não inclua nenhum texto, símbolos ou explicações, apenas os números." }, { inline_data: { mime_type: "image/jpeg", data: base64Image } } ] }] }; const geminiResponse = await fetch(geminiUrl, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(requestBody) }); const geminiData = await geminiResponse.json(); // NOVO: Verifica se a API do Google retornou um erro (ex: chave inválida) if (geminiData.error) { throw new Error(`Erro da API: ${geminiData.error.message}`); } // NOVO: Verifica se a IA encontrou alguma resposta válida if (!geminiData.candidates || !geminiData.candidates[0].content) { throw new Error("A IA não reconheceu comida nesta imagem. Tente novamente."); } // Extrai o número da resposta do Gemini const textoResposta = geminiData.candidates[0].content.parts[0].text.trim(); const caloriasEstimadas = parseInt(textoResposta); if (isNaN(caloriasEstimadas)) throw new Error(`A IA não retornou um número válido. Resposta: ${textoResposta}`); statusMsg.innerText = `Identificado: ${caloriasEstimadas} kcal. A guardar na planilha...`; // 2. Enviar as calorias para o Google Sheets via Apps Script await fetch(APPS_SCRIPT_URL, { method: 'POST', body: JSON.stringify({ caloria: caloriasEstimadas }), mode: 'no-cors' }); statusMsg.innerText = "Salvo com sucesso!"; setTimeout(atualizarTotalDiario, 1500); } catch (error) { console.error("Erro detalhado:", error); statusMsg.innerText = "Erro: " + error.message; } finally { btnCapturar.disabled = false; }
